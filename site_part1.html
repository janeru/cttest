dateplot<-function(x){
  fname<-paste0(x,".csv")
  d <- read.csv(fname, header=TRUE)
  d$newdate <- d$date
  d$newtime<- d$time
  d$finaldate<- paste0(d$newdate," ",d$newtime)
  dn <- select(d,gps_lat,gps_lon,s_d0,finaldate)
  tmp <- data.frame(dn$finaldate,dn$s_d0,dn$gps_lat,dn$gps_lon)
  colnames(tmp) <- c("date","pm25","lat","lon")
  tmp$date <- as.POSIXct(strptime(tmp$date, format= "%Y-%m-%d %H:%M:%S", "GMT"))
  return (calendarPlot(tmp, pollutant = "pm25",annotate = "date",year = 2017, breaks = c(0,15.4,35.4,54.4,150.4,250.4,350.4), labels = c("0.0 - 15.4","15.5 - 35.4","35.5 - 54.4","54.5 - 150.4","150.5 - 250.4",">=250.5")
                       ,cols = c("Green", "Yellow", "orange", "Red", "Purple","brown")))
}
hoursplot<-function(x){
  fname<-paste0(x,".csv")
  d <- read.csv(fname, header=TRUE)
  d$newdate <- d$date
  d$newtime<- d$time
  d$finaldate<- paste0(d$newdate," ",d$newtime)
  dn <- select(d,gps_lat,gps_lon,s_d0,finaldate)
  tmp <- data.frame(dn$finaldate,dn$s_d0,dn$gps_lat,dn$gps_lon)
  colnames(tmp) <- c("date","pm25","lat","lon")
  tmp$date <- as.POSIXct(strptime(tmp$date, format= "%Y-%m-%d %H:%M:%S", "GMT"))
  tmp$nmonth<-substr(tmp$date,6,7)
  tmp$ndate<-substr(tmp$date,9,10)
  tmp$nhour<-substr(tmp$date,12,13)
  tmp$nmonth<-as.numeric(tmp$nmonth)
  tmp$nhour<-as.numeric(tmp$nhour)
  tmp$ndate<-as.numeric(tmp$ndate)
  return(tmp)
}
